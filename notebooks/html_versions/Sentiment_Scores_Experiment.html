def vader_by_line(lyrics):
    split = lyrics.splitlines()
    lines = list(filter(None, split))
    scores = []
    for line in lines:
        score = analyser.polarity_scores(line)
        scores.append(score['compound'])
    scores = list(filter(lambda x: x != 0.0, scores))
    return np.mean(scores)
